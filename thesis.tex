\if@delimsize
\else
  \mathopen{
    \left.{\kern-\nulldelimiterspace}

    \middle#1
    {\kern-\nulldelimiterspace}\right.
  }
\fi
}

\def\DeclareDelimiterX#1[#2]#3#4#5 {
  \DeclareRobustCommand{#1}[#2]{
    \if@delimsize
      #5
    \else
      \begingroup
        \mathopen{}
        \@delimsizetrue
        \@delimiterWrapper{\left#3}{
          \vphantom{#5}
          {\kern-\nulldelimiterspace}
        }{\right.}
        \def\@delimcontent{#5}
        \@delimsizefalse
        {#5}
        \mathclose{
        \@delimsizetrue
        \@delimiterWrapper{\left.}{
          {\kern-\nulldelimiterspace}
          \vphantom{#5}
        }{\right#4}
        }
      \endgroup
    \fi
  }
}
